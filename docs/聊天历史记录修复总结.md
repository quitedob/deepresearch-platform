```go
func (api *ChatAPI) GetSessions(c *gin.Context) {
    // ... 获取会话列表
    
    for i, s := range sessions {
        // 获取最后一条消息作为预览
        var lastMessage string
        if api.chatDAO != nil {
            lastMsg, err := api.chatDAO.GetLastMessageBySessionID(c.Request.Context(), s.ID)
            if err == nil && lastMsg != nil {
                // 截取前100个字符作为预览
                if len(lastMsg.Content) > 100 {
                    lastMessage = lastMsg.Content[:100] + "..."
                } else {
                    lastMessage = lastMsg.Content
                }
            }
        }
        
        sessionResponses[i] = response.NewChatSessionResponseWithLastMessage(s, lastMessage)
    }
}
```

**功能**:
- 为每个会话查询最后一条消息
- 截取前100个字符作为预览
- 超过100字符添加 "..." 省略号

## 深度研究历史记录

### 问题
深度研究（Research）也需要历史记录，但目前：
- ✅ 有 `ResearchSession` 模型
- ✅ 有 `ResearchDAO`
- ✅ 有 `/api/v1/research/sessions` 接口
- ⚠️ 前端可能需要统一的历史记录视图

### 建议方案

#### 方案 1: 分离显示
- 聊天历史：显示普通对话
- 研究历史：显示深度研究任务
- 优点：清晰分离
- 缺点：用户需要切换视图

#### 方案 2: 统一显示
- 在历史列表中同时显示聊天和研究
- 使用 `model_type` 区分：
  - `default` - 普通聊天
  - `deep` - 深度思考聊天
  - `research` - 深度研究
- 优点：统一体验
- 缺点：需要合并两个 API 的数据

### 推荐实现（方案 2）

**前端 Store**:
```javascript
async fetchHistoryList() {
  try {
    // 获取聊天会话
    const chatResponse = await chatAPI.getSessions()
    const chatSessions = (chatResponse.sessions || []).map(s => ({
      ...s,
      type: 'chat',
      last_message: s.last_message || ''
    }))
    
    // 获取研究会话
    const researchResponse = await researchAPI.getResearchSessions()
    const researchSessions = (researchResponse.sessions || []).map(s => ({
      id: s.id,
      title: s.query,
      type: 'research',
      status: s.status,
      progress: s.progress,
      message_count: 0,
      last_message: `研究状态: ${s.status}`,
      updated_at: new Date(s.updated_at),
      created_at: new Date(s.created_at)
    }))
    
    // 合并并按时间排序
    this.historyList = [...chatSessions, ...researchSessions]
      .sort((a, b) => b.updated_at - a.updated_at)
  } catch (error) {
    console.error('获取历史列表失败:', error)
  }
}
```

## 数据库查询性能优化

### 当前实现
```go
// 为每个会话单独查询最后一条消息
for i, s := range sessions {
    lastMsg, _ := api.chatDAO.GetLastMessageBySessionID(ctx, s.ID)
    // ...
}
```

**问题**: N+1 查询问题，如果有20个会话，需要执行21次数据库查询

### 优化方案

#### 方案 1: 批量查询
```go
// 在 ChatDAO 中添加批量查询方法
func (c *ChatDAO) GetLastMessagesBySessionIDs(ctx context.Context, sessionIDs []string) (map[string]*model.Message, error) {
    var messages []*model.Message
    
    // 使用子查询获取每个会话的最后一条消息
    err := c.db.WithContext(ctx).Raw(`
        SELECT DISTINCT ON (session_id) *
        FROM messages
        WHERE session_id = ANY($1)
        ORDER BY session_id, created_at DESC
    `, pq.Array(sessionIDs)).Scan(&messages).Error
    
    if err != nil {
        return nil, err
    }
    
    // 转换为 map
    result := make(map[string]*model.Message)
    for _, msg := range messages {
        result[msg.SessionID] = msg
    }
    
    return result, nil
}
```

#### 方案 2: 在 ChatSession 中缓存
```go
// 在 ChatSession 模型中添加字段
type ChatSession struct {
    // ... 原有字段
    LastMessageContent string    `gorm:"-" json:"-"` // 不存储到数据库
    LastMessageAt      time.Time `json:"last_message_at"`
}

// 在创建消息时更新
func (c *ChatDAO) CreateMessage(ctx context.Context, message *model.Message) error {
    return c.db.WithContext(ctx).Transaction(func(tx *gorm.DB) error {
        // 创建消息
        if err := tx.Create(message).Error; err != nil {
            return err
        }
        
        // 更新会话的最后消息时间
        if err := tx.Model(&model.ChatSession{}).
            Where("id = ?", message.SessionID).
            Update("last_message_at", message.CreatedAt).Error; err != nil {
            return err
        }
        
        return nil
    })
}
```
