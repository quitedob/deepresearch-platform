### 数据库迁移注意事项

由于添加了新的模型字段，需要执行数据库迁移：

```sql
-- 为 chat_sessions 表添加 version 字段
ALTER TABLE chat_sessions ADD COLUMN IF NOT EXISTS version INTEGER DEFAULT 1;

-- 为 messages 表添加 version, updated_at, deleted_at 字段
ALTER TABLE messages ADD COLUMN IF NOT EXISTS version INTEGER DEFAULT 1;
ALTER TABLE messages ADD COLUMN IF NOT EXISTS updated_at TIMESTAMP;
ALTER TABLE messages ADD COLUMN IF NOT EXISTS deleted_at TIMESTAMP;
CREATE INDEX IF NOT EXISTS idx_messages_deleted_at ON messages(deleted_at);

-- 创建研究证据表
CREATE TABLE IF NOT EXISTS research_evidences (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    research_id UUID NOT NULL REFERENCES research_sessions(id),
    task_id UUID REFERENCES research_tasks(id),
    source_type VARCHAR(50) NOT NULL,
    source_url TEXT,
    source_title TEXT,
    content TEXT,
    relevance REAL DEFAULT 0,
    credibility REAL DEFAULT 0,
    metadata JSONB,
    is_verified BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IF NOT EXISTS idx_research_evidences_research_id ON research_evidences(research_id);

-- 创建研究引用表
CREATE TABLE IF NOT EXISTS research_citations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    research_id UUID NOT NULL REFERENCES research_sessions(id),
    evidence_id UUID REFERENCES research_evidences(id),
    citation_type VARCHAR(50) NOT NULL,
    position INTEGER DEFAULT 0,
    text TEXT,
    formatted_ref TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IF NOT EXISTS idx_research_citations_research_id ON research_citations(research_id);

-- 创建研究发现表
CREATE TABLE IF NOT EXISTS research_findings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    research_id UUID NOT NULL REFERENCES research_sessions(id),
    category VARCHAR(50) NOT NULL,
    title TEXT,
    content TEXT NOT NULL,
    confidence REAL DEFAULT 0,
    importance INTEGER DEFAULT 0,
    evidence_ids JSONB,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IF NOT EXISTS idx_research_findings_research_id ON research_findings(research_id);
```

或者使用 GORM 自动迁移（在 main.go 中已配置）。
